### 10.5 Pattern в‘Ө: Reverse Engineering

```javascript
// PROVE: class is indeed just sugar over prototype!

class Animal {
  constructor(name) {
    this.name = name;
  }
  speak() {
    return `${this.name} makes a noise`;
  }
}

// Proof 1: typeof
console.log(typeof Animal); // "function" вҖ” NOT "class"!

// Proof 2: has .prototype
console.log(Animal.prototype.speak); // Ж’ speak()

// Proof 3: prototype chain works identically
const dog = new Animal("Rex");
console.log(dog.__proto__ === Animal.prototype); // true
console.log(dog.__proto__.__proto__ === Object.prototype); // true

// Proof 4: can add methods to .prototype manually
Animal.prototype.run = function () {
  return `${this.name} runs!`;
};
console.log(dog.run()); // "Rex runs!" вң…

// Proof 5: Object.getOwnPropertyNames
console.log(Object.getOwnPropertyNames(Animal.prototype));
// ["constructor", "speak", "run"]
// вҶ’ Methods are ON .prototype, just like Solution 3!
```
