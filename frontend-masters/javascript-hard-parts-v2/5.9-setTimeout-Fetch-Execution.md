  }
}

setTimeout(printHello, 0);

const futureData = fetch("https://jsonplaceholder.typicode.com/todos/1");
futureData.then((response) => response.json()).then(display);

blockFor300ms();

console.log("Me first!");

// PREDICTED OUTPUT (we'll verify in next lesson!):
// 1. "Me first!"         ← sync code first!
// 2. { ... todo data }   ← Promise callback (MICROTASK!)
// 3. "Hello"             ← setTimeout callback (TASK!)
// → Promise .then BEFORE setTimeout!
// → MICROTASK queue > CALLBACK queue!

```javascript
// CHỨNG MINH 2: Combined trace simulator

class AsyncSystemSimulator {
  constructor() {
    this.memory = {};
    this.callStack = ["global()"];
    this.callbackQueue = [];
    this.microtaskQueue = []; // SPOILER!
    this.browserTimers = [];
    this.browserNetwork = [];
    this.console = [];
    this.time = 0;
    this.globalDone = false;
  }

  log(msg) {
    console.log(`  [${this.time}ms] ${msg}`);
  }

  declareFunction(name) {
    this.memory[name] = `ƒ ${name}`;
    this.log(`MEMORY: ${name} = ƒ`);
  }

  // setTimeout — ONE-pronged (browser only!)
  setTimeout(callback, delay) {
    this.log(`setTimeout(${callback}, ${delay}ms) → BROWSER!`);
    this.log(`  Timer: ${delay}ms → complete!`);
    this.callbackQueue.push(callback);
    this.log(`  ${callback} → CALLBACK QUEUE!`);
  }

  // fetch — TWO-pronged!
  fetch(url) {
    this.log(`fetch("${url}") → TWO PRONGS!`);

    // PRONG 1 — JS: Promise object!
    const promise = {
      value: undefined,
      onFulfilled: [],
      status: "pending",
    };
    this.log(`  PRONG ① JS: Promise { value: undefined }`);

    // PRONG 2 — Browser: Network request!
    this.log(`  PRONG ② Browser: Network → ${url} (GET)`);
    this.browserNetwork.push({
      url,
      promise,
      completesAt: 200, // simulate 200ms response
    });

    return promise;
  }

  then(promise, fn) {
    promise.onFulfilled.push(fn);
    this.log(`.then(${fn.name || "fn"}) → pushed to onFulfilled!`);
  }

  blockFor(ms) {
    this.callStack.push(`blockFor${ms}ms()`);
    this.log(`CALL: blockFor${ms}ms() → stack: [${this.callStack.join("→")}]`);
    this.log(`  ...blocking for ${ms}ms...`);

    // Check if network completes during blocking:
    this.browserNetwork.forEach((req) => {
      if (req.completesAt <= this.time + ms && req.status !== "complete") {
        const completeTime = req.completesAt;
        this.log(`  [${completeTime}ms] 🌐 Network response!`);
        req.promise.value = "cute puppy";
        req.promise.status = "fulfilled";
        req.status = "complete";
        // Trigger functions → MICROTASK queue!
        req.promise.onFulfilled.forEach((fn) => {
          this.microtaskQueue.push(fn.name || "display");
          this.log(
            `  [${completeTime}ms] ${fn.name || "display"} → MICROTASK QUEUE!`,
          );
        });
      }
    });

    this.time += ms;
    this.callStack.pop();
    this.log(`POP: blockFor${ms}ms() done`);
  }

  consoleLog(msg) {
    this.console.push({ time: this.time, msg });
    this.log(`console.log("${msg}")`);
  }

  finishGlobal() {
    this.globalDone = true;
    this.log(`\n═══ ALL GLOBAL CODE DONE! ═══`);
    this.log(`Event loop checking...`);

    // MICROTASK queue first!
    if (this.microtaskQueue.length > 0) {
      this.log(`\nMICROTASK QUEUE (HIGHER PRIORITY!):`);
      while (this.microtaskQueue.length > 0) {
        const fn = this.microtaskQueue.shift();
        this.time++;
        this.log(`  DEQUEUE: ${fn}() → call stack!`);
        this.console.push({ time: this.time, msg: `(${fn} output)` });
      }
    }

    // THEN callback queue!
    if (this.callbackQueue.length > 0) {
      this.log(`\nCALLBACK QUEUE (LOWER PRIORITY!):`);
      while (this.callbackQueue.length > 0) {
        const fn = this.callbackQueue.shift();
        this.time++;
        this.log(`  DEQUEUE: ${fn}() → call stack!`);
        this.console.push({ time: this.time, msg: `(${fn} output)` });
      }
    }
  }

  printConsole() {
    console.log("\n═══ FINAL CONSOLE OUTPUT ═══");
    this.console.forEach((e) => console.log(`  [${e.time}ms] ${e.msg}`));
  }
}

// RUN THE SIMULATION:
console.log("═══ COMBINED TRACE SIMULATION ═══\n");
const sim = new AsyncSystemSimulator();

// Declare functions:
sim.declareFunction("display");
sim.declareFunction("printHello");
sim.declareFunction("blockFor300ms");

// setTimeout(printHello, 0):
sim.setTimeout("printHello", 0);

// fetch("tiktok.com/will"):
sim.time = 1;
const futureData = sim.fetch("tiktok.com/will");

// futureData.then(display):
sim.then(futureData, { name: "display" });

// blockFor300ms():
sim.time = 2;
sim.blockFor(300);

// console.log("Me first!"):
sim.time = 302;
sim.consoleLog("Me first!");

// Global done!
sim.finishGlobal();
sim.printConsole();
```

## §9. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 9.1 Pattern ①: 5 Whys — Combined Trace

```
5 WHYS: TẠI SAO CẦN TRACE CẢ setTimeout VÀ fetch?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao trace cả hai cùng lúc?
  └→ Vì cả hai TẠO CALLBACKS!
     setTimeout → callback queue!
     fetch.then → ??? queue!
     Cần biết PRIORITY khi cả hai ready!

  WHY ②: Tại sao thêm blockFor300ms?
  └→ Vì cần CHỨNG MINH tất cả phải chờ sync!
     300ms blocking = đủ lâu để:
     → Timer complete (0ms = instant!)
     → Network response arrive (~200ms!)
     → CẢ HAI callbacks ready nhưng PHẢI CHỜ!

  WHY ③: Tại sao printHello vào callback queue?
  └→ Vì setTimeout = old-school, ONE-pronged!
     Browser timer feature → callback queue!
     = Task queue!

  WHY ④: Tại sao display KHÔNG vào callback queue?
  └→ Vì fetch = new-school, TWO-pronged!
     Promise.then callbacks → MICROTASK queue!
     = DIFFERENT queue! HIGHER priority!
     → SẼ REVEAL ở bài tiếp theo!

  WHY ⑤: Tại sao cần 2 queues?
  └→ Vì Promises CẦN higher priority!
     Promise = tracking work in JS!
     Cần response NHANH khi data arrives!
     Microtask queue = runs BEFORE task queue!
```

### 9.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — TWO-PRONGED vs ONE-PRONGED:
═══════════════════════════════════════════════════════════════

  ONE-PRONGED (setTimeout, addEventListener):
  ┌──────────────────────────────────────────────────┐
  │ → Sets up browser work ONLY!                    │
  │ → Returns nothing useful in JS!                 │
  │ → Callback → CALLBACK QUEUE (task queue!)       │
  │ → Lower priority!                               │
  │                                                  │
  │ JS: "I have NO IDEA what's happening in browser" │
  └──────────────────────────────────────────────────┘

  TWO-PRONGED (fetch, Promise-based APIs):
  ┌──────────────────────────────────────────────────┐
  │ → Sets up browser work + returns Promise in JS!  │
  │ → Promise = tracking object!                    │
  │ → .then callbacks → MICROTASK QUEUE!            │
  │ → Higher priority!                              │
  │                                                  │
  │ JS: "I know EXACTLY what's going on!"           │
  └──────────────────────────────────────────────────┘

  WHY DIFFERENT QUEUES?
  ECMAScript Spec:
  → Promise reactions = "PromiseJobs"
  → PromiseJobs = enqueued as microtasks
  → Microtasks run BEFORE next macrotask
  → = Tighter integration with JS execution!
```

### 9.5 Pattern ⑤: Reverse Engineering

```javascript
// PROVING: setTimeout vs Promise priority

console.log("Sync 1");

setTimeout(() => console.log("setTimeout"), 0);

Promise.resolve().then(() => console.log("Promise.then"));

console.log("Sync 2");

// Output:
// "Sync 1"          ← sync!
// "Sync 2"          ← sync!
// "Promise.then"    ← MICROTASK! (runs FIRST!)
// "setTimeout"      ← TASK! (runs AFTER!)

// PROOF:
// → Promise.then ALWAYS before setTimeout!
// → Even though BOTH are 0ms delay!
// → Microtask queue > callback queue!
```

### 9.6 Pattern ⑥: Lịch Sử

```
LỊCH SỬ: QUEUE PRIORITY — WHY MICROTASKS?
═══════════════════════════════════════════════════════════════

  Pre-2015: Only ONE queue!
  │     → All callbacks = same priority!
  │     → FIFO = first come, first served!
  │     → Simple but inflexible!
  │
  ↓
  2015: ES2015 — Promises introduced!
  │     → Need: Promise reactions should be FAST!
  │     → Solution: SEPARATE queue for promises!
  │     → Microtask queue = higher priority!
  │     → HTML spec updated: processing model
  │       includes "perform a microtask checkpoint"
  │       AFTER each task!
  │
  ↓
  2016: MutationObserver also uses microtasks!
  │     → DOM mutations → microtask queue!
  │     → = Consistent with Promise behavior!
  │
  ↓
  2019: queueMicrotask() API!
  │     → Direct access to microtask queue!
  │     → queueMicrotask(() => { ... })
  │     → No need to create a Promise just
  │       for microtask scheduling!
  │
  ↓
  2024: Well-established two-queue model!
        → Microtask = promises, mutations
        → Task (macro) = timers, events, I/O
        → Convention now ~10 years old!
```
