// CHỨNG MINH 3: Implement ToPrimitive algorithm

function toPrimitive(obj, hint = "default") {
  // Step 1: Check for @@toPrimitive (Symbol.toPrimitive)
  if (typeof obj[Symbol.toPrimitive] === "function") {
    const result = obj[Symbol.toPrimitive](hint);
    if (typeof result !== "object") return result;
    throw new TypeError("@@toPrimitive must return primitive!");
  }

  // Step 2: No @@toPrimitive → fallback chain
  if (hint === "number" || hint === "default") {
    // Try valueOf() first, then toString()
    if (typeof obj.valueOf === "function") {
      const val = obj.valueOf();
      if (typeof val !== "object") return val;
    }
    if (typeof obj.toString === "function") {
      const str = obj.toString();
      if (typeof str !== "object") return str;
    }
  }

  if (hint === "string") {
    // Try toString() first, then valueOf()
    if (typeof obj.toString === "function") {
      const str = obj.toString();
      if (typeof str !== "object") return str;
    }
    if (typeof obj.valueOf === "function") {
      const val = obj.valueOf();
      if (typeof val !== "object") return val;
    }
  }

  throw new TypeError("Cannot convert to primitive!");
}

// Test with regular object:

    return 42;
  },

    return "hello";
  },
};

console.log(toPrimitive(myObj, "string")); // "hello" (toString first!)
console.log(toPrimitive(myObj, "default")); // 42 (valueOf first!)

traceToPrimitive(plain, "number");
// ❌ No @@toPrimitive → valueOf() → 42 ✅

traceToPrimitive(plain, "string");
// ❌ No @@toPrimitive → toString() → "forty-two" ✅

## §10. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 10.1 Pattern ①: 5 Whys — ToPrimitive

```
5 WHYS: TẠI SAO TOPRIMITIVE TỒN TẠI?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao operators cần convert objects?
  └→ Vì operators (+, -, <, ==) chỉ BIẾT primitives!
     Objects = complex → cần "rút" primitive value ra!

  WHY ②: Tại sao cần @@toPrimitive thay vì chỉ valueOf?
  └→ Vì valueOf KHÔNG CHO PHÉP phân biệt hint!
     @@toPrimitive("number") vs ("string") = khác kết quả!
     valueOf() = luôn trả 1 giá trị duy nhất!

  WHY ③: Tại sao Date cần riêng @@toPrimitive?
  └→ Vì Date MUỐN: + → string, - → number!
     Mặc định: + → valueOf (number)!
     Date override: + → toString (date string)!
     → date1 + date2 = "Mon Jan 15..." (human readable!)
     → date1 - date2 = 1000 (math!)

  WHY ④: Tại sao @@toPrimitive dùng Symbol, không string?
  └→ Vì backward compatibility!
     Nếu dùng string "toPrimitive" → có thể clash
     với property names cũ!
     Symbol = GUARANTEED UNIQUE = no clash!

  WHY ⑤: Tại sao phải chờ 20 năm (1995→2015)?
  └→ Vì Symbols chưa tồn tại trước ES2015!
     Symbols = prerequisite cho @@toPrimitive!
     Cả 2 được thiết kế CÙNG LÚC trong ES2015!
```

### 10.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — ECMASCRIPT SPEC:
═══════════════════════════════════════════════════════════════

  7.1.1 ToPrimitive(input, preferredType):

  1. If input = primitive → RETURN input (done!)

  2. Let exoticToPrim = GetMethod(input, @@toPrimitive)
     → @@toPrimitive = Symbol.toPrimitive

  3. If exoticToPrim is NOT undefined:
     → Let hint = preferredType → "number"/"string"/"default"
     → Let result = Call(exoticToPrim, input, [hint])
     → If result is primitive → RETURN result
     → Else → throw TypeError!

  4. If exoticToPrim IS undefined (fallback!):
     → OrdinaryToPrimitive(input, hint):
       → If hint="number": try valueOf(), then toString()
       → If hint="string": try toString(), then valueOf()

  PRIORITY:
  @@toPrimitive (nếu có) > valueOf/toString (fallback)!

  → @@toPrimitive = HIGH PRIORITY override!
  → Nếu define → valueOf/toString BỊ BỎ QUA!
```

  ├──────────────────┼──────────────────┼──────────────────┤

### 10.5 Pattern ⑤: Reverse Engineering

```javascript
// TỰ BUILD: Custom object with @@toPrimitive

// Use case: Currency object
const usd = {
  amount: 42.5,
  currency: "USD",

  [Symbol.toPrimitive](hint) {
    switch (hint) {
      case "number":
        return this.amount; // for math!
      case "string":
        return `$${this.amount.toFixed(2)} ${this.currency}`;
      default: // "default" — used by == and +
        return this.amount;
    }
  },
};

// Math (hint = "number"):
console.log(usd * 2); // 85 (42.50 * 2)
console.log(usd - 10); // 32.5

// String (hint = "string"):
console.log(`${usd}`); // "$42.50 USD"
console.log(String(usd)); // "$42.50 USD"

// Default (hint = "default"):
console.log(usd + 10); // 52.5 (amount + 10)
console.log(usd == 42.5); // true!

// Comparison:
const eur = {
  amount: 38.0,
  currency: "EUR",
  [Symbol.toPrimitive](hint) {
    if (hint === "number") return this.amount;
    return `€${this.amount.toFixed(2)}`;
  },
};

console.log(usd - eur); // 4.5 (42.50 - 38.00)
console.log(usd > eur); // true (42.50 > 38.00)
```
