## §2. Joe's Q: Browser References JS Memory — How?

>

## §4. Chris's Q: Promise Trong Local Context — Heap Persistence!

> Chris: _"Can you do this in a temporary execution context? Those objects have a life outside?"_
>
> Will: _"Any object is stored as a reference to its underlying position in the heap. Even if we lose the name in JavaScript, the object still remains in that heap as long as the WEB BROWSER references it."_

```
HEAP PERSISTENCE — BROWSER KEEPS IT ALIVE!:
═══════════════════════════════════════════════════════════════

  function makeRequest() {
    const localPromise = fetch("tiktok.com/videos");
    localPromise.then(display);
    // return (nothing)
  }

  makeRequest();
  // makeRequest EC destroyed!
  // localPromise label = GONE!

  BUT:
  ┌──────────────────────────────────────────────────┐
  │ HEAP:                                            │
  │ ┌────────────────────────────────────────────┐   │
  │ │ Promise object at addr 0x4F               │   │
  │ │ STILL EXISTS!                             │   │
  │ │                                            │   │
  │ │ Why? Because:                              │   │
  │ │ → Browser network request REFERENCES it!   │   │
  │ │ → Browser keeps the reference alive!       │   │
  │ │ → Object won't be garbage collected!       │   │
  │ │ → Even though JS label is gone!            │   │
  │ │                                            │   │
  │ │ When response arrives:                     │   │
  │ │ → Browser uses its reference to find obj!  │   │
  │ │ → Updates result property!                 │   │
  │ │ → Triggers fulfillReactions!               │   │
  │ │ → display(data) runs!                      │   │
  │ └────────────────────────────────────────────┘   │
  │                                                  │
  │ After response handled:                          │
  │ → Browser drops reference                       │
  │ → No more references to object!                 │
  │ → Garbage collector can clean up! 🗑️            │
  └──────────────────────────────────────────────────┘

  Will: "References to stuff in JavaScript, made
  not even from within JavaScript, but BY THE WEB
  BROWSER, persist the existence of the promise object."

  GIỐNG CLOSURE!
  ┌──────────────────────────────────────────────────┐
  │ Closure: function giữ reference → data persists  │
  │ Promise: browser giữ reference → object persists │
  │                                                  │
  │ CẢ HAI: reference prevents garbage collection!  │
  │ CẢ HAI: data sống lâu hơn scope tạo ra nó!     │
  └──────────────────────────────────────────────────┘
```

```javascript
// CHỨNG MINH 2: Simulate browser reference to JS heap

class JSHeap {
  constructor() {
    this.objects = new Map(); // addr → object
    this.nextAddr = 0x100;
  }

  allocate(obj) {
    const addr = `0x${this.nextAddr.toString(16).toUpperCase()}`;
    this.nextAddr += 0x10;
    this.objects.set(addr, obj);
    console.log(`[HEAP] Allocated at ${addr}:`, JSON.stringify(obj));
    return addr;
  }

  get(addr) {
    return this.objects.get(addr);
  }

  update(addr, prop, value) {
    const obj = this.objects.get(addr);
    if (obj) {
      obj[prop] = value;
      console.log(`[HEAP] Updated ${addr}.${prop} = "${value}"`);
    }
  }
}

class BrowserRuntime {
  constructor(heap) {
    this.heap = heap;
    this.networkRequests = [];
  }

  fetch(url) {
    // PRONG 1: Allocate Promise in JS heap:
    const promiseAddr = this.heap.allocate({
      result: undefined,
      fulfillReactions: [],
      status: "pending",
    });
    console.log(`[FETCH] Promise at ${promiseAddr}`);

    // PRONG 2: Network request (browser maintains REFERENCE!):
    const request = {
      url,
      promiseAddr, // ← REFERENCE to JS heap!
      complete: false,
    };
    this.networkRequests.push(request);
    console.log(`[BROWSER] Network → ${url}`);
    console.log(`[BROWSER] Keeps REFERENCE to ${promiseAddr}!`);

    // Simulate async response:
    setTimeout(() => {
      console.log(`\n[BROWSER 270ms] Response from ${url}!`);
      // Browser uses its REFERENCE to update JS heap:
      this.heap.update(promiseAddr, "result", "cute puppy");
      this.heap.update(promiseAddr, "status", "fulfilled");

      // Trigger fulfillReactions:
      const promise = this.heap.get(promiseAddr);
      promise.fulfillReactions.forEach((fn) => {
        console.log(`[AUTO] Running ${fn.name}("${promise.result}")`);
        fn(promise.result);
      });
    }, 270);

    return promiseAddr;
  }

  then(promiseAddr, fn) {
    const promise = this.heap.get(promiseAddr);
    promise.fulfillReactions.push(fn);
    console.log(`[.then] Pushed ${fn.name} into fulfillReactions`);
  }
}

// Simulate:
console.log("═══ BROWSER ↔ JS HEAP REFERENCE ═══\n");
const heap = new JSHeap();
const browser = new BrowserRuntime(heap);

function display(data) {
  console.log(`  >>> "${data}" displayed!`);
}

// fetch creates Promise in heap + network request with reference:
const addr = browser.fetch("tiktok.com/will");

// .then pushes into fulfillReactions:
browser.then(addr, display);

console.log('\n[1ms] console.log("Me first!")');
console.log("  >>> Me first!\n");
console.log("[1ms] Global done! Waiting for browser...");
// 270ms later → browser uses reference → updates heap → runs display!
```

```javascript
// CHỨNG MINH 4: Heap persistence — closure analogy

function createFetcher(url) {
  // Local scope:
  const promise = fetch(url);
  const handlers = [];

  promise
    .then((r) => r.json())
    .then((data) => {
      handlers.forEach((fn) => fn(data));
    });

  // Return function (closure!) but NOT the promise:
  return function addHandler(fn) {
    handlers.push(fn);
  };
}

const addHandler = createFetcher(
  "https://jsonplaceholder.typicode.com/todos/1",
);

addHandler((data) => console.log("Handler 1:", data.title));

// → Handlers persisted by closure!
```

## §8. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 8.1 Pattern ①: 5 Whys — Browser Reference

```
5 WHYS: TẠI SAO BROWSER CÓ THỂ ACCESS JS MEMORY?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao browser cần access JS memory?
  └→ Vì cần UPDATE Promise.result khi data arrives!
     Không có access → không thể fill data!

  WHY ②: Tại sao dùng reference, không copy?
  └→ Vì copy = 2 objects riêng biệt!
     Update copy ≠ update original!
     Reference = SAME object = update real Promise!

  WHY ③: Tại sao browser CÓ THỂ reference JS heap?
  └→ Vì JS engine (V8) CHẠY TRONG browser!
     Chrome owns V8 → can access V8's memory!
     = "Implementation detail"!

  WHY ④: Tại sao gọi là "implementation detail"?
  └→ Vì SPEC không nói HOW to connect!
     Spec chỉ nói WHAT should happen!
     Mỗi browser implement khác nhau!
     Chrome (V8), Firefox (SpiderMonkey), Safari (JSC)!

  WHY ⑤: Tại sao promise persist sau EC destroy?
  └→ Vì browser reference = prevents GC!
     Garbage collector: "có reference → don't delete!"
     Same rule as closure!
```

### 8.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — GARBAGE COLLECTION:
═══════════════════════════════════════════════════════════════

  Mark-and-Sweep Algorithm (V8):

  1. Find all "roots" (references that are always live):
     → Global variables
     → Call stack variables
     → ★ Browser API references ← THIS IS KEY!

  2. Mark everything reachable from roots

  3. Sweep (delete) everything NOT marked

  PROMISE IN LOCAL SCOPE:
  ┌──────────────────────────────────────────────────┐
  │ function fn() {                                  │
  │   const p = fetch(url);   // promise created!    │
  │   p.then(display);                               │
  │ }                                                │
  │ fn();                                            │
  │ // fn() EC destroyed → p label GONE!            │
  │                                                  │
  │ BUT: Browser network request = ROOT reference!   │
  │ → Promise object marked as REACHABLE!           │
  │ → GC won't sweep it!                            │
  │ → Persists until response handled!              │
  │                                                  │
  │ After response:                                  │
  │ → Browser drops reference                       │
  │ → No more roots reference this object            │
  │ → GC sweeps → memory freed! 🗑️                  │
  └──────────────────────────────────────────────────┘
```

### 8.5 Pattern ⑤: Reverse Engineering

```javascript
// TỰ BUILD: Reference counting / GC simulation

class GarbageCollector {
  constructor() {
    this.heap = new Map();
    this.roots = new Set();
  }

  allocate(id, obj) {
    this.heap.set(id, { obj, refCount: 0 });
    return id;
  }

  addRoot(id, source) {
    const entry = this.heap.get(id);
    if (entry) {
      entry.refCount++;
      this.roots.add(id);
      console.log(
        `[GC] Root added: ${source} → ${id} (refs: ${entry.refCount})`,
      );
    }
  }

  removeRoot(id, source) {
    const entry = this.heap.get(id);
    if (entry) {
      entry.refCount--;
      console.log(
        `[GC] Root removed: ${source} → ${id} (refs: ${entry.refCount})`,
      );
      if (entry.refCount <= 0) {
        this.roots.delete(id);
        this.heap.delete(id);
        console.log(`[GC] 🗑️ Object ${id} garbage collected!`);
      }
    }
  }
}

const gc = new GarbageCollector();

// Fetch creates Promise:
gc.allocate("promise_0x4F", { result: undefined });

// 2 references:
gc.addRoot("promise_0x4F", "JS:futureData"); // JS label
gc.addRoot("promise_0x4F", "Browser:network"); // Browser ref

// EC destroyed → JS label gone:
gc.removeRoot("promise_0x4F", "JS:futureData");
// → refCount = 1 → NOT collected! Browser still needs it!

// Response arrives → browser done:
console.log("\n[BROWSER] Response arrived! Work done!");
gc.removeRoot("promise_0x4F", "Browser:network");
// → refCount = 0 → 🗑️ COLLECTED!
```
