```javascript
// CHỨNG MINH 2: Full callback queue simulation

class EventLoopSimulator {
  constructor() {
    this.callStack = ["global()"];
    this.callbackQueue = [];
    this.consoleOutput = [];
    this.time = 0;
    this.globalCodeDone = false;
  }

  log(msg) {
    console.log(`  [${this.time}ms] ${msg}`);
  }

  // Facade function:
  setTimeout(callback, delay) {
    this.log(`setTimeout(${callback.name}, ${delay}ms) → BROWSER!`);

    // Browser timer (simulated):
    if (delay === 0) {
      this.log(`  Timer: 0ms → COMPLETE immediately!`);
      this.log(`  BUT: goes to CALLBACK QUEUE, not stack!`);
      this.callbackQueue.push({
        name: callback.name,
        fn: callback,
        readyAt: this.time,
      });
    } else {
      // For non-zero delays, would schedule later
      this.callbackQueue.push({
        name: callback.name,
        fn: callback,
        readyAt: this.time + delay,
      });
    }
  }

  consoleLog(message) {
    this.consoleOutput.push({ time: this.time, message });
    this.log(`console.log("${message}")`);
  }

  callFunction(name, fn, duration) {
    this.callStack.push(`${name}()`);
    this.log(`CALL: ${name}() → stack: [${this.callStack.join("→")}]`);

    // Check if callback tries to interrupt:
    if (this.callbackQueue.length > 0) {
      this.log(`  Callback queue: "${this.callbackQueue[0].name}" says:`);
      this.log(`  "Excuse me! Can I come back on??" 🥺`);
      this.log(`  ANSWER: ❌ NO! ${name} is running!`);
    }

    // Simulate execution time:
    this.time += duration;
    this.log(`  ...running for ${duration}ms...`);

    // Pop off stack:
    this.callStack.pop();
    this.log(`POP: ${name}() done → stack: [${this.callStack.join("→")}]`);
  }

  finishGlobalCode() {
    this.globalCodeDone = true;
    this.log(`ALL GLOBAL CODE DONE! ✅`);
    this.log(`Checking callback queue...`);
    this.processCallbackQueue();
  }

  processCallbackQueue() {
    while (this.callbackQueue.length > 0) {
      const cb = this.callbackQueue.shift();
      this.time++;
      this.log(
        `DEQUEUE: ${cb.name}() → call stack! (waited ${this.time - cb.readyAt}ms!)`,
      );
      this.callStack.push(`${cb.name}()`);
      cb.fn(this);
      this.callStack.pop();
      this.log(`POP: ${cb.name}() done`);
    }
  }
}

// Run the lesson's code:
console.log("═══ CALLBACK QUEUE SIMULATION ═══\n");
const sim = new EventLoopSimulator();

// Line 1-8: declare functions
sim.log("declare printHello, blockFor1Sec");

// Line 10: setTimeout(printHello, 0)
sim.setTimeout({ name: "printHello" }, 0);

// Line 12: blockFor1Sec()
sim.callFunction("blockFor1Sec", () => {}, 1000);

// Line 14: console.log("Me first!")
sim.consoleLog("Me first!");

// Global code done → process queue
sim.finishGlobalCode();

console.log("\n═══ FINAL CONSOLE OUTPUT ═══");
sim.consoleOutput.forEach((e) => console.log(`  [${e.time}ms] ${e.message}`));
```

## §9. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 9.1 Pattern ①: 5 Whys — Callback Queue

```
5 WHYS: TẠI SAO CẦN CALLBACK QUEUE?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao callback không lên call stack ngay?
  └→ Vì JS single-threaded! Nếu nhảy vào giữa chừng
     → code đang chạy bị INTERRUPT!
     → Race condition! Unpredictable!

  WHY ②: Tại sao cần queue (hàng đợi)?
  └→ Vì có THỂ nhiều callbacks cùng ready!
     setTimeout 0, setTimeout 0, click event...
     Queue = FIFO = first-come first-served!
     = Deterministic order!

  WHY ③: Tại sao phải chờ ALL global code?
  └→ Vì "half-done" global code = DANGEROUS!
     Imagine: đang gán biến → callback thay đổi biến!
     RUN-TO-COMPLETION guarantee = safe!

  WHY ④: Tại sao không async JS bằng cách khác?
  └→ Vì single-threaded = KHÔNG THỂ interrupt!
     Multi-threaded = possible nhưng complex + races!
     Queue + event loop = elegant middle ground!

  WHY ⑤: Tại sao setTimeout(fn, 0) ≠ "chạy ngay"?
  └→ Vì 0ms = chỉ "hoàn thành timer" nhanh!
     Callback vẫn phải qua queue → event loop!
     Event loop chỉ check khi global code DONE!
     → Minimum delay = KHÔNG guaranteed!
```

### 9.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — EVENT LOOP SPEC:
═══════════════════════════════════════════════════════════════

  HTML Living Standard — 8.1.7.3 Processing Model:

  "An event loop must continually run through
  the following steps:

  1. Let oldestTask be null.
  2. Let taskStartTime be the current high
     resolution time.
  3. IF the task queue has a runnable task:
     → Set oldestTask to that task
     → Remove it from task queue
     → Set running task to oldestTask
     → Run oldestTask's steps
     → Set running task back to null
  4. Perform a microtask checkpoint
  5. Update the rendering
  6. GOTO step 1"

  KEY WORD: "runnable task"
  → Task CHỈ runnable khi execution context
     stack chỉ còn empty hoặc global!
  → = Will's rule: "all global code must finish!"

  RUN-TO-COMPLETION:
  → 8.1.4: "Each task is associated with a
     source and must be run to completion"
  → = KHÔNG interrupt giữa chừng!
```

### 9.5 Pattern ⑤: Reverse Engineering

```javascript
// TỰ BUILD: Event loop with callback queue

class EventLoop {
  constructor() {
    this.callStack = [];
    this.callbackQueue = [];
    this.running = false;
  }

  // Push synchronous work:
  pushSync(taskName, fn) {
    this.callStack.push(taskName);
    console.log(`[STACK] PUSH: ${taskName} → [${this.callStack.join(",")}]`);
    fn();
    this.callStack.pop();
    console.log(`[STACK] POP: ${taskName} → [${this.callStack.join(",")}]`);

    // After each sync task, check queue:
    this._tick();
  }

  // Enqueue async callback:
  enqueue(callbackName, fn) {
    this.callbackQueue.push({ name: callbackName, fn });
    console.log(`[QUEUE] Enqueue: ${callbackName} → waiting...`);
  }

  // Event loop tick:
  _tick() {
    if (this.callStack.length === 0 && !this.running) {
      while (this.callbackQueue.length > 0) {
        const cb = this.callbackQueue.shift();
        console.log(`[EVENT LOOP] Stack empty! Dequeue: ${cb.name}`);
        this.callStack.push(cb.name);
        cb.fn();
        this.callStack.pop();
      }
    }
  }

  // Mark global code as running:
  startGlobal(fn) {
    this.running = true;
    this.callStack.push("global()");
    console.log("\n[START] Global execution begins");
    fn();
    this.callStack.pop();
    this.running = false;
    console.log("[END] Global execution complete!\n");
    this._tick(); // NOW check the queue!
  }
}

// Simulate the lesson:
const loop = new EventLoop();

loop.startGlobal(() => {
  // setTimeout(printHello, 0) → complete instantly → queue!
  loop.enqueue("printHello", () => console.log("Hello"));

  // blockFor1Sec():
  console.log("[JS] blockFor1Sec() running...");
  // (simulated — just a log)

  // console.log("Me first!")
  console.log("[JS] Me first!");
});

// After global: event loop processes queue!
// Output:
// [QUEUE] Enqueue: printHello → waiting...
// [JS] blockFor1Sec() running...
// [JS] Me first!
// [END] Global execution complete!
// [EVENT LOOP] Stack empty! Dequeue: printHello
// Hello
```
