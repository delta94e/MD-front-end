function myNew(Constructor, ...args) {
  console.log("🔵 STEP ①: Create empty object!");

// CHỨNG MINH 4: Why .prototype replaces functionStore

// The old way (manual store):
const manualStore = {
  greet() {
    console.log(`Hi ${this.name}`);
  },
};
function createUser(name) {
  const user = Object.create(manualStore);
  user.name = name;
  return user;

function CreateUser(name) {
  this.name = name;
}
CreateUser.prototype.greet = function () {
  console.log(`Hi ${this.name}`);

const userOld = createUser("Ari");
const userNew = new CreateUser("Ari");

userNew.greet(); // "Hi Ari" ✅

console.log(userOld.__proto__ === manualStore); // true

## §10. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 10.1 Pattern ①: 5 Whys — Full new Trace

```
5 WHYS: TẠI SAO new WORKS?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao new tạo object tự động?
  └→ Vì EVERY constructor call cần 1 object!
     100% of the time → nên automate!

  WHY ②: Tại sao link đến .prototype mà không elsewhere?
  └→ Vì .prototype đã CÓ SẴN trên every function!
     JS biết chính xác nơi để tìm shared methods!
     Không cần separate store declaration!

  WHY ③: Tại sao label là "this" mà không tên khác?
  └→ Vì... JavaScript design decision (questionable!)
     Will (sarcastic): "Genius design."
     Cùng keyword, khác nghĩa hoàn toàn!

  WHY ④: Tại sao auto return?
  └→ Vì constructor LUÔN return the object!
     Nếu bạn return primitive → bị bỏ qua!
     Nếu bạn return object → thay thế this!

  WHY ⑤: Tại sao "nothing changed under the hood"?
  └→ Vì new chỉ là AUTOMATION! Không phải new concept!
     Cùng prototype chain! Cùng __proto__! Cùng lookup!
     Chỉ bớt viết code! "3 letters, 3 steps!"
```

### 10.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — new EXECUTION:
═══════════════════════════════════════════════════════════════

  ECMAScript [[Construct]](argumentsList):

  1. Let thisArgument = OrdinaryCreateFromConstructor(F)
     → Creates: {} with [[Prototype]] = F.prototype

  2. Let result = OrdinaryCallBindThis(F, thisArgument)
     → Sets this = thisArgument
     → Runs function body

  3. If Type(result) is Object → return result
     Else → return thisArgument

  KEY INSIGHT:
  → Step 1 = BLUE STEP ① (create object)
  → Step 1 also = BLUE STEP ② (set __proto__)
  → Step 2 = MANUAL work (this.name = name)
  → Step 3 = BLUE STEP ③ (auto return this)
```

### 10.5 Pattern ⑤: Reverse Engineering

// Solution 2:
const store = {
  increment() {
    this.score++;
  },
};

  const user = Object.create(store);
  user.name = name;
  user.score = score;
  return user;
}

// Solution 3:
function UserCreator(name, score) {
  this.name = name;
  this.score = score;
}
UserCreator.prototype.increment = function () {
  this.score++;
};

const u2 = createUser("Ari", 3);
const u3 = new UserCreator("Ari", 3);

// Same structure:

console.log(u2.score === u3.score); // true

// Same chain behavior:

u3.increment();

console.log(u2.__proto__ === store); // true
console.log(u3.__proto__ === UserCreator.prototype); // true
