```javascript
function display(data) {

function blockFor300ms() {

setTimeout(printHello, 0);
```

```javascript
// CHỨNG MINH 2: Build a dual-queue event loop!

class DualQueueEventLoop {
  constructor() {
    this.callStack = [];
    this.microtaskQueue = []; // ★ Priority ①!
    this.callbackQueue = []; // Priority ②!
    this.console = [];
    this.time = 0;
    this.globalDone = false;
  }

  log(msg) {
    console.log(`  [${String(this.time).padStart(3)}ms] ${msg}`);
  }

  // Sync: push to call stack:
  execute(name, duration = 0) {
    this.callStack.push(name);
    this.log(`CALL: ${name} → stack: [${this.callStack.join(" → ")}]`);

    if (duration > 0) {
      this.log(`  ...blocking for ${duration}ms...`);
      // Check for microtask arrivals during blocking:
      this._checkBrowserDuringBlock(duration);
      this.time += duration;
    }

    this.callStack.pop();
    this.log(`POP: ${name}`);
  }

  // Console.log:
  consoleLog(msg) {
    this.console.push({ time: this.time, msg });
    this.log(`CONSOLE: "${msg}"`);
  }

  // setTimeout → callback queue (ONE-pronged):
  setTimeoutSim(fnName, delay) {
    this.log(`setTimeout(${fnName}, ${delay}ms) → BROWSER`);
    if (delay === 0) {
      this.log(`  Timer complete → ${fnName} → CALLBACK QUEUE!`);
      this.callbackQueue.push(fnName);
    }
  }

  // fetch → microtask queue (TWO-pronged):
  fetchSim(url, thenFn, responseTime, responseData) {
    this.log(`fetch("${url}") → TWO PRONGS!`);
    this.log(`  PRONG ①: Promise { value: undefined }`);
    this.log(`  PRONG ②: Network → ${url}`);
    this.log(`  .then(${thenFn}) → stored in onFulfilled!`);

    // Store for later when response arrives:
    this._pendingNetwork = {
      fn: thenFn,
      data: responseData,
      completesAt: responseTime,
    };
  }

  _checkBrowserDuringBlock(duration) {
    if (this._pendingNetwork) {
      const net = this._pendingNetwork;
      if (net.completesAt <= this.time + duration) {
        this.log(`  [${net.completesAt}ms] 🌐 Response: "${net.data}"!`);
        this.log(`  [${net.completesAt}ms] → value = "${net.data}"`);
        this.log(`  [${net.completesAt}ms] → ${net.fn} → ★ MICROTASK QUEUE!`);
        this.microtaskQueue.push({ fn: net.fn, data: net.data });
        this._pendingNetwork = null;
      }
    }
  }

  // Run the event loop after global is done!
  runEventLoop() {
    this.log(`\n═══ GLOBAL CODE DONE! EVENT LOOP STARTS! ═══\n`);

    // STEP 1: Drain ALL microtasks FIRST!
    if (this.microtaskQueue.length > 0) {
      this.log(`★ MICROTASK QUEUE CHECK (PRIORITY ①!):`);
      while (this.microtaskQueue.length > 0) {
        const task = this.microtaskQueue.shift();
        this.time++;
        this.log(`  DEQUEUE: ${task.fn}("${task.data}") → call stack!`);
        this.callStack.push(`${task.fn}("${task.data}")`);
        this.consoleLog(task.data);
        this.callStack.pop();
        this.log(`  POP: ${task.fn}`);
      }
    }

    // STEP 2: THEN callback queue!
    if (this.callbackQueue.length > 0) {
      this.log(`\nCALLBACK QUEUE CHECK (PRIORITY ②!):`);
      while (this.callbackQueue.length > 0) {
        const fnName = this.callbackQueue.shift();
        this.time++;
        this.log(`  DEQUEUE: ${fnName}() → call stack! (FINALLY! 😭)`);
        this.callStack.push(`${fnName}()`);
        this.consoleLog("Hello");
        this.callStack.pop();
        this.log(`  POP: ${fnName}`);
      }
    }

    // Print final console:
    console.log("\n═══ FINAL CONSOLE OUTPUT ═══");
    this.console.forEach((e, i) =>
      console.log(`  ${i + 1}. "${e.msg}"  [${e.time}ms]`),
    );
  }
}

// RUN THE FULL SIMULATION:
console.log("═══ DUAL-QUEUE SIMULATION ═══\n");

const loop = new DualQueueEventLoop();

// setTimeout(printHello, 0):
loop.setTimeoutSim("printHello", 0);

// fetch("tiktok.com/will"):
loop.time = 1;
loop.fetchSim("tiktok.com/will", "display", 270, "cute puppy");

// blockFor300ms():
loop.time = 2;
loop.execute("blockFor300ms", 300);

// console.log("Me first!"):
```

## §11. 🔬 Deep Analysis Patterns — 6 Tư Duy Phân Tích Sâu

### 11.1 Pattern ①: 5 Whys — Microtask Queue

```
5 WHYS: TẠI SAO CẦN MICROTASK QUEUE RIÊNG?
═══════════════════════════════════════════════════════════════

  WHY ①: Tại sao không dùng chung callback queue?
  └→ Vì Promise cần HIGHER PRIORITY!
     Promise = tracking work trong JS!
     Data arrived → cần xử lý NGAY!
     Không nên chờ sau setTimeout!

  WHY ②: Tại sao Promise cần priority?
  └→ Vì Promise là TWO-PRONGED!
     JS đã tạo placeholder object!
     Object CẦN được update ASAP!
     Code depending on data NEEDS to run!

  WHY ③: Tại sao drain ALL microtasks trước ANY task?
  └→ Vì consistency and predictability!
     All Promise reactions run together!
     No interleaving with setTimeout callbacks!
     = Deterministic behavior!

  WHY ④: Tại sao Event Loop check microtask FIRST?
  └→ Vì HTML spec mandates it!
     "Perform a microtask checkpoint"
     runs AFTER each task completes!
     = Built into browser event loop spec!

  WHY ⑤: Tại sao setTimeout 0ms chạy SAU Promise.then?
  └→ Vì 2 DIFFERENT queues!
     setTimeout → task queue (lower priority!)
     Promise.then → microtask queue (HIGHER!)
     microtask ALWAYS checked first!
```

### 11.2 Pattern ②: First Principles

```
FIRST PRINCIPLES — HTML SPEC EVENT LOOP:
═══════════════════════════════════════════════════════════════

  HTML Living Standard — Event Loop Processing Model:

  1. Let task = dequeue from task queue
  2. Execute task
  3. ★ PERFORM A MICROTASK CHECKPOINT:
     a. While microtask queue is NOT empty:
        i.  Let microtask = dequeue from microtask queue
        ii. Execute microtask
        iii. (may add MORE microtasks!)
     b. Done when queue empty!
  4. Update rendering (if needed)
  5. Go to step 1!

  KEY: Step 3 = DRAIN ALL microtasks!
  → This is WHY microtasks always run first!
  → This is WHY all microtasks complete before
     the next task from the task queue!
```

### 11.6 Pattern ⑥: Lịch Sử

```
LỊCH SỬ: MICROTASK QUEUE ORIGIN:

  │     → All callbacks = same queue!
  │     → Simple FIFO!
  │     → No priority system!
  │
  ↓

  │     → First use of "microtasks" in browsers!
  │     → DOM mutations needed immediate handling!
  │     → Browsers added internal microtask queue!
  │
  ↓
  2015: ES2015 — Promises native!
  │     → ECMAScript spec: "PromiseJobs"
  │     → HTML spec: run as "microtasks"
  │     → Two-queue model standardized!
  │     → Will: "Promise deferred functionality
  │       gets put into a SEPARATE queue!"
  │
  ↓
  2019: queueMicrotask() API!
  │     → Direct access to microtask queue!
  │     → No need to create throwaway Promise!
  │     → queueMicrotask(() => { ... })
  │
  ↓
  2024: Well-established model:
```
